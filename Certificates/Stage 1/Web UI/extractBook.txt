using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ExtractBookCode      //Do not change the namespace name
{
    public class Program      //Do not change the class name
    {
        public static void Main(String[] arg)         //Do not change the method signature
        {
            Console.WriteLine("Enter the book code of length ");  
            string code = Console.ReadLine();
            
            if (code.Length == 18)
            {
                string depCode = code.Substring(0, 3);
                string pubYear = code.Substring(3, 4);
                string noPages = code.Substring(7, 5);
                string bookId = code.Substring(12,6);
                
                if(depCode == "101" || depCode == "102" || depCode == "103")
                    Console.WriteLine("Department Code        : {0}", depCode);
                else
                    Console.WriteLine("Invalid Department Code");
                    
                int year = Convert.ToInt32(pubYear);
                if(year>=1900 && year<=2020)
                    Console.WriteLine("Year of Publication     : {0}", year);
                else
                    Console.WriteLine("Invalid Year ");
                    
                long pages = Int64.Parse(noPages);
                if(pages >= 00001 && pages<=99999)
                    Console.WriteLine("Number of Pages        : {0}", noPages);
                else
                    Console.WriteLine("Invalid Page Numbers");
                    
                char ch = Convert.ToChar(code.Substring(12,1));
                if(Char.IsLetter(ch))
                {
                    bool flag = true;
                    for(int i=0; i<5; i++){
                        if(!Char.IsNumber(Convert.ToChar(code.Substring(13+i,1))))
                            flag = false;
                    }
                    if(flag)
                    {
                        Console.WriteLine("Book ID                      : {0}", bookId);
                    }
                    else
                    {
                        Console.WriteLine("Invalid Book ID");
                    }
                }
                else
                {
                    Console.WriteLine("Invalid Book ID");
                }
            }
            else
                Console.WriteLine("Invalid Book Code ");
        }
            
    }
}
