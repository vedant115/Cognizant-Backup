            function calMonthlyRentalCost(){
                let schemeCost = {
                    "Local": 200,
                    "STD": 350,
                    "FullTalkTime": 500
                }
                let monRental = {
                    "Pre": 75,
                    "Post": 150
                }
                let duration = parseInt(document.getElementById("duration").value);
                let schemes = document.getElementsByName("scheme");
                let scheme;
                for(let op of schemes){
                    if(op.checked){
                        scheme = op.value;
                        break;
                    }
                }
                let conType = document.getElementById("connectiontype").value;
                let totalCost = (duration * schemeCost[scheme]) + monRental[conType];
                document.getElementById("result").innerHTML = "The Total Monthly Rental Cost is Rs. " + totalCost;
            }
        </script>
